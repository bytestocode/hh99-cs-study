## 콜백지옥

### 동기 vs 비동기
코드의 흐름이 연속적이냐 불연속적이냐
```javascript
const log = console.log;

log('1');
setTimeout(() => log('2'), 3000);
log('3');

// 동기 콜백
function printImmediately(f) {
  f();
}
printImmediately(() => log('sync callback'));

// 비동기 콜백
function printWithDelay(f, delay) {
  setTimeout(f, delay);
}
printWithDelay(() => log('async callback'), 1000);

```
